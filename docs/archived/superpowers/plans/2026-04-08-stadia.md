- [ ] **Step 6: Implement pitch.ts**

Create `packages/stadia/src/geometry/pitch.ts`:

This file exports `computePitchMarkings(crop)` which returns an array of marking objects. Each marking has an `id`, `type` (line | rect | circle | arc), and geometric coordinates in meters relative to pitch origin (0,0 = top-left of full pitch in vertical orientation).

Key marking definitions using PITCH constants:

- **boundary**: rect at (0, 0, PITCH.width, PITCH.length)
- **halfway**: line at y = PITCH.length / 2
- **center-circle**: circle at (PITCH.width/2, PITCH.length/2, PITCH.centerCircleRadius)
- **penalty-area-top**: rect centered at top, width=penaltyAreaWidth, depth=penaltyAreaLength
- **goal-area-top**: rect centered at top, width=goalAreaWidth, depth=goalAreaLength
- **penalty-spot-top**: circle at (PITCH.width/2, penaltySpotDistance, 0.3)
- **penalty-arc-top**: arc portion of circle centered at penalty spot (PITCH.width/2, penaltySpotDistance), radius=penaltyArcRadius. The arc endpoints are where this circle intersects the penalty area boundary line (y=penaltyAreaLength). Compute the x-offset as `sqrt(penaltyArcRadius² - (penaltyAreaLength - penaltySpotDistance)²)` = `sqrt(9.15² - 5.5²)` = `sqrt(53.47)` ≈ 7.31m. Do NOT copy the ShotMap's approximation which uses 9.15m as the x-offset — that produces a semicircle, not the correct intersection arc
- **corner arcs**: quarter-circle at each corner, radius=cornerArcRadius

For half crop, include only the attacking-end markings (top of the pitch) plus a center arc.
For penalty-area crop, include only the penalty area and its contents.

Read `packages/react/src/ShotMap.tsx` lines 169-225 to see how the current half-pitch SVG is constructed — use it as a structural reference but fix the penalty arc geometry (see arc intersection note above).

```typescript
import { PITCH } from "../geometry/constants.js";

export type ProjectFn = (x: number, y: number) => { x: number; y: number };
export type PitchCrop = "full" | "half" | "penalty-area";
export type Orientation = "vertical" | "horizontal";

/**
 * Create a projection function that maps Campos 0-100 coordinates
 * to SVG user units (meter-scale) for a given crop and orientation.
 *
 * Always returns coordinates — SVG viewport handles clipping.
 * Returned values are in meters (1 SVG user unit = 1 meter).
 */
export function createPitchProjection(
  crop: PitchCrop,
  orientation: Orientation,
): ProjectFn {
  return (camposX: number, camposY: number) => {
    // Campos: x 0-100 (own goal → opposition goal), y 0-100 (left → right)
    // SVG meters: x = pitch width dimension, y = pitch length dimension
    // In vertical orientation: SVG x = Campos y, SVG y = inverted Campos x
    const meterX = (camposY / 100) * PITCH.width;
    const meterY = ((100 - camposX) / 100) * PITCH.length;

    if (orientation === "horizontal") {
      // Horizontal: SVG x = Campos x direction, SVG y = Campos y direction
      const hx = (camposX / 100) * PITCH.length;
      const hy = ((100 - camposY) / 100) * PITCH.width;
      return { x: hx, y: hy };
    }

    return { x: meterX, y: meterY };
  };
}
```

Note: The projection maps the full 0-100 range to the full pitch dimensions regardless of crop. The crop only affects the SVG viewBox (which determines what's visible), not the projection math. This means out-of-crop coordinates produce valid SVG positions that are simply outside the viewport.

```typescript
import { GOAL } from "../geometry/constants.js";

export type GoalProjectFn = (goalY: number, goalZ: number) => { x: number; y: number };
export type GoalFacing = "striker" | "goalkeeper";

/**
 * Create a projection for goal-mouth coordinates.
 * goalY: 0-100 across goal width (left post to right post)
 * goalZ: 0-100 ground to crossbar
 *
 * Striker facing: left post = screen-left
 * Goalkeeper facing: mirrored (left post = screen-right)
 */
export function createGoalProjection(facing: GoalFacing): GoalProjectFn {
  return (goalY: number, goalZ: number) => {
    const normalizedX = goalY / 100;
    const normalizedZ = goalZ / 100;

    const svgX =
      facing === "striker" ? normalizedX * GOAL.width : (1 - normalizedX) * GOAL.width;
    const svgY = (1 - normalizedZ) * GOAL.depth;

    return { x: svgX, y: svgY };
  };
}
```

```typescript
import { PITCH, GOAL } from "../geometry/constants.js";
import type { PitchCrop, Orientation } from "./pitch-transform.js";

export type ViewBox = {
  minX: number;
  minY: number;
  width: number;
  height: number;
};

type Padding = number | { top: number; right: number; bottom: number; left: number };

function resolvePadding(
  padding: Padding | undefined,
  referenceWidth: number,
): { top: number; right: number; bottom: number; left: number } {
  if (padding == null) return { top: 0, right: 0, bottom: 0, left: 0 };
  if (typeof padding === "number") {
    const p = (padding / 100) * referenceWidth;
    return { top: p, right: p, bottom: p, left: p };
  }
  return {
    top: (padding.top / 100) * referenceWidth,
    right: (padding.right / 100) * referenceWidth,
    bottom: (padding.bottom / 100) * referenceWidth,
    left: (padding.left / 100) * referenceWidth,
  };
}

export function computeViewBox(
  surface: PitchCrop | "goal",
  orientation: Orientation,
  padding?: Padding,
): ViewBox {
  let baseWidth: number;
  let baseHeight: number;
  let baseMinX = 0;
  let baseMinY = 0;

  if (surface === "goal") {
    baseWidth = GOAL.width;
    baseHeight = GOAL.depth;
  } else if (surface === "full") {
    baseWidth = orientation === "vertical" ? PITCH.width : PITCH.length;
    baseHeight = orientation === "vertical" ? PITCH.length : PITCH.width;
  } else if (surface === "half") {
    baseWidth = orientation === "vertical" ? PITCH.width : PITCH.length / 2;
    baseHeight = orientation === "vertical" ? PITCH.length / 2 : PITCH.width;
  } else {
    // penalty-area: centered on the 18-yard box with some padding
    const paWidth = PITCH.penaltyAreaWidth + 4; // 2m padding each side
    const paHeight = PITCH.penaltyAreaLength + 4;
    baseWidth = orientation === "vertical" ? paWidth : paHeight;
    baseHeight = orientation === "vertical" ? paHeight : paWidth;
    if (orientation === "vertical") {
      baseMinX = (PITCH.width - paWidth) / 2;
      baseMinY = 0 - 2; // 2m above goal line
    } else {
      baseMinX = 0 - 2; // 2m before goal line
      baseMinY = (PITCH.width - paWidth) / 2;
    }
  }

  const refWidth = orientation === "vertical" ? PITCH.width : PITCH.length;
  const pad = resolvePadding(padding, refWidth);

  return {
    minX: baseMinX - pad.left,
    minY: baseMinY - pad.top,
    width: baseWidth + pad.left + pad.right,
    height: baseHeight + pad.top + pad.bottom,
  };
}
```

```typescript
// Hardcoded rather than imported to avoid cross-package JSON import complexity.
// If tokens change, update these values to match.
const DARK_PITCH: Required<PitchColors> = {
  fill: "#12141a", // color.surface.hero
  lines: "#2a2e38", // color.border.hairline
  markings: "#2a2e3880", // color.border.hairline at 50% opacity
};

  fill: "#f5f5f7",
  lines: "#d1d5db",
  markings: "#d1d5db80",
};

const DARK_GOAL: Required<GoalColors> = {
  frame: "#f5f5f7",
  net: "#12141a4d",
  ground: "#2a2e38",
};

const LIGHT_GOAL: Required<GoalColors> = {
  frame: "#1a1a1a",
  net: "#f5f5f74d",
  ground: "#d1d5db",
};

/** Filter out undefined values from an overrides object to prevent spread from
 *  overwriting base values with undefined (relevant with exactOptionalPropertyTypes). */
function stripUndefined<T extends Record<string, unknown>>(obj: T): Partial<T> {
  const result: Record<string, unknown> = {};
  for (const [key, value] of Object.entries(obj)) {
    if (value !== undefined) result[key] = value;
  }
  return result as Partial<T>;
}

export function resolvePitchColors(
  theme: Theme,
  overrides?: PitchColors,
): Required<PitchColors> {
  const base = theme === "dark" ? DARK_PITCH : LIGHT_PITCH;
  return { ...base, ...(overrides ? stripUndefined(overrides) : {}) };
}

export function resolveGoalColors(
  theme: Theme,
  overrides?: GoalColors,
): Required<GoalColors> {
  const base = theme === "dark" ? DARK_GOAL : LIGHT_GOAL;
  return { ...base, ...(overrides ? stripUndefined(overrides) : {}) };
}

export type { PitchColors, GoalColors, Theme };
```

```tsx
import { useMemo, type ReactNode, type CSSProperties } from "react";
import {
  createPitchProjection,
  type PitchCrop,
  type Orientation,
  type ProjectFn,
} from "../transforms/pitch-transform.js";
import { computeViewBox } from "../transforms/viewbox.js";
import { resolvePitchColors, type PitchColors, type Theme } from "./theme.js";
import { PitchMarkings } from "./PitchMarkings.js";
import { TacticalMarkings } from "./TacticalMarkings.js";

type PitchMarkingsConfig = {
  halfSpaces?: boolean;
  thirds?: boolean;
  zones?: boolean;
};

type Padding = number | { top: number; right: number; bottom: number; left: number };

export type PitchProps = {
  crop: PitchCrop;
  orientation?: Orientation;
  theme?: Theme;
  markings?: PitchMarkingsConfig;
  colors?: PitchColors;
  padding?: Padding;
  interactive?: boolean;
  className?: string;
  style?: CSSProperties;
  children: (ctx: { project: ProjectFn }) => ReactNode;
};

export function Pitch({
  crop,
  orientation = "vertical",
  theme = "dark",
  markings,
  colors: colorOverrides,
  padding,
  interactive = true,
  className,
  style,
  children,
}: PitchProps) {
  const project = useMemo(
    () => createPitchProjection(crop, orientation),
    [crop, orientation],
  );

  const viewBox = useMemo(
    () => computeViewBox(crop, orientation, padding),
    [crop, orientation, padding],
  );

  const colors = useMemo(
    () => resolvePitchColors(theme, colorOverrides),
    [theme, colorOverrides],
  );

  const vbString = `${viewBox.minX} ${viewBox.minY} ${viewBox.width} ${viewBox.height}`;

  return (
    <svg
      viewBox={vbString}
      overflow="hidden"
      preserveAspectRatio="xMidYMid meet"
      className={className}
      style={{ width: "100%", height: "100%", ...style }}
      {...(!interactive ? { pointerEvents: "none" as const } : {})}
    >
      <PitchMarkings crop={crop} orientation={orientation} colors={colors} />
      {markings && (
        <TacticalMarkings
          crop={crop}
          orientation={orientation}
          markings={markings}
          color={colors.markings}
        />
      )}
      {children({ project })}
    </svg>
  );
}
```
